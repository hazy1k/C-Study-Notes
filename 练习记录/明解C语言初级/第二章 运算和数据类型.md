{
    float a;
    scanf("%f", &a); // 注意啦，这里不能写%d了，因为是浮点数

    printf("你输入的是：%.6f", a);

    return 0;
}

## 练习2.5

![](https://img.picgo.net/2024/05/20/-2024-05-20-215433521d875f32c6694a.png)

```c
#include <stdio.h>
int main()
{
    int a, b;
    printf("请输入两个整数.\n");
    scanf("%d", &a);  
    scanf("%d", &b);  

    printf("整数A：%d\n", a);
    printf("整数B：%d\n", b);
    
    double c = (double)a / b * 100;  
    printf("A是B的%lf%%\n", c); // %% 表示百分号

    return 0;
}
```
